Dijkstra's Code


#include <iostream>
#include <vector>
#include <queue>
#include <climits>
#include <unordered_map>
#include <set>
#include <cstdlib>

using namespace std;

struct Edge {
    int destination;
    int weight;
};

struct Node {
    int vertex;
    int distance;
    bool operator>(const Node& other) const {
        return distance > other.distance;
    }
};

class CityGraph {
public:
    CityGraph(int vertices) {
        V = vertices;
        adjList = (vector<Edge>*)malloc(V * sizeof(vector<Edge>));
        dist = (int*)malloc(V * sizeof(int));
        parent = (int*)malloc(V * sizeof(int));

        for (int i = 0; i < V; ++i) {
            dist[i] = INT_MAX;
            parent[i] = -1;
        }
    }

    ~CityGraph() {
        free(adjList);
        free(dist);
        free(parent);
    }

    void addEdge(int u, int v, int weight) {
        adjList[u].push_back({v, weight});
        adjList[v].push_back({u, weight});
    }

    void run(int start) {
        dist[start] = 0;

        priority_queue<Node, vector<Node>, greater<Node>> pq;
        pq.push({start, 0});

        set<int> visited;

        while (!pq.empty()) {
            int u = pq.top().vertex;
            pq.pop();
            visited.insert(u);

            for (const auto& edge : adjList[u]) {
                int v = edge.destination;
                int weight = edge.weight;

                if (visited.find(v) == visited.end() && dist[u] + weight < dist[v]) {
                    dist[v] = dist[u] + weight;
                    parent[v] = u;
                    pq.push({v, dist[v]});
                }
            }
        }
    }

    void printShortestRoute(int target) {
        if (dist[target] == INT_MAX) {
            cout << "No route found from source to " << target << endl;
            return;
        }

        cout << "Shortest route distance from source to " << target << " is " << dist[target] << endl;
        cout << "Route: ";
        printPath(target);
        cout << endl;
    }

private:
    int V;
    vector<Edge>* adjList;
    int* dist;
    int* parent;

    void printPath(int v) {
        if (parent[v] == -1) {
            cout << v;
            return;
        }
        printPath(parent[v]);
        cout << " -> " << v;
    }
};

int main() {
    int V = 6;
    CityGraph city(V);

    city.addEdge(0, 1, 10);
    city.addEdge(0, 2, 15);
    city.addEdge(1, 3, 12);
    city.addEdge(1, 4, 15);
    city.addEdge(2, 4, 10);
    city.addEdge(3, 5, 10);
    city.addEdge(4, 5, 5);

    city.run(0);

    for (int i = 0; i < V; ++i) {
        city.printShortestRoute(i);
    }

    return 0;

Greedy (Linear Programming) Code


#include <iostream>
#include <glpk.h>
#include <vector>

using namespace std;

int main() {
    int n;
    cout << "Enter the number of energy sources: ";
    cin >> n;

    vector<pair<int, int>> energySources(n);

    for (int i = 0; i < n; ++i) {
        cout << "Enter energy generated (in kWh) and cost (in thousands of dollars) for energy source " << i + 1 << ": ";
        cin >> energySources[i].first >> energySources[i].second;
    }

    int totalBudget;
    cout << "Enter total budget (in thousands): ";
    cin >> totalBudget;

    glp_prob *lp;
    lp = glp_create_prob();
    glp_set_prob_name(lp, "Energy Placement Optimization");
    glp_set_obj_dir(lp, GLP_MAX);

    n = energySources.size();

    glp_add_cols(lp, n);
    for (int i = 1; i <= n; ++i) {
        glp_set_col_kind(lp, i, GLP_BV);
        glp_set_obj_coef(lp, i, energySources[i - 1].first);
    }

    glp_add_rows(lp, 1);
    glp_set_row_bnds(lp, 1, GLP_UP, 0.0, totalBudget);

    vector<int> costCoefficients(n, 0);
    for (int i = 0; i < n; ++i) {
        costCoefficients[i] = energySources[i].second;
    }
    glp_set_mat_row(lp, 1, n, costCoefficients.data());

    glp_simplex(lp, NULL);

    cout << "Selected Energy Sources for Maximum Energy Generation:" << endl;
    int totalEnergy = 0;
    for (int i = 1; i <= n; ++i) {
        if (glp_get_col_prim(lp, i) > 0.5) {
            cout << "Energy Source " << i << ": " << energySources[i - 1].first << " kWh, Cost: " << energySources[i - 1].second << " thousand dollars" << endl;
            totalEnergy += energySources[i - 1].first;
        }
    }

    cout << "Maximum energy generated: " << totalEnergy << " kWh" << endl;

    glp_delete_prob(lp);

    return 0;

Prim's Code


#include <iostream>
#include <vector>
#include <climits>
#include <queue>
using namespace std;

struct Edge {
    int dest, weight;
};

struct Node {
    int vertex, key;
    bool operator>(const Node &other) const {
        return key > other.key;
    }
};

class Graph {
public:
    Graph(int vertices) {
        V = vertices;
        adjList.resize(vertices);
    }

    void addEdge(int u, int v, int weight) {
        adjList[u].push_back({v, weight});
        adjList[v].push_back({u, weight});
    }

    void primMST() {
        vector<int> key(V, INT_MAX);
        vector<int> parent(V, -1);
        vector<bool> inMST(V, false);

        priority_queue<Node, vector<Node>, greater<Node>> pq;
        key[0] = 0;
        pq.push({0, 0});

        while (!pq.empty()) {
            int u = pq.top().vertex;
            pq.pop();
            inMST[u] = true;

            for (const auto &edge : adjList[u]) {
                int v = edge.dest;
                int weight = edge.weight;
                if (!inMST[v] && weight < key[v]) {
                    key[v] = weight;
                    pq.push({v, key[v]});
                    parent[v] = u;
                }
            }
        }

        printMST(parent);
    }

private:
    int V;
    vector<vector<Edge>> adjList;

    void printMST(const vector<int> &parent) {
        cout << "Edge \tWeight\n";
        for (int i = 1; i < V; ++i) {
            cout << parent[i] << " - " << i << "\t" << findWeight(parent[i], i) << endl;
        }
    }

    int findWeight(int u, int v) {
        for (const auto &edge : adjList[u]) {
            if (edge.dest == v)
                return edge.weight;
        }
        return 0;
    }
};

int main() {
    int V, E;
    cout << "Enter the number of vertices: ";
    cin >> V;

    cout << "Enter the number of edges: ";
    cin >> E;

    Graph graph(V);

    cout << "Enter the edges (u v weight):\n";
    for (int i = 0; i < E; ++i) {
        int u, v, weight;
        cin >> u >> v >> weight;
        graph.addEdge(u, v, weight);
    }

    cout << "\nMinimum Spanning Tree using Prim's Algorithm:\n";
    graph.primMST();

    return 0;
