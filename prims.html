Prim's Algorithm

Algorithm

ALGORITHM Prim(G)
// Prim’s algorithm to construct a minimum spanning tree
// Input: A weighted connected graph G(V, E)
// Output: ET, the set of edges composing the MST of G
VT ← {v₀}
ET ← Ø
for i ← 1 to |V| - 1 do
    find a minimum weight edge e* = (v*, u*) along all the edges (v, u) such that
    v is in VT and u is in V – VT
    VT ← VT ∪ {u*}
    ET ← ET ∪ {e*}
return ET

Prim's Algorithm Implementation in C++

#include <iostream>
#include <vector>
#include <climits>
#include <queue>
using namespace std;

struct Edge {
    int dest, weight;
};

struct Node {
    int vertex, key;
    bool operator>(const Node &other) const {
        return key > other.key;
    }
};

class Graph {
public:
    Graph(int vertices) {
        V = vertices;
        adjList.resize(vertices);
    }

    void addEdge(int u, int v, int weight) {
        adjList[u].push_back({v, weight});
        adjList[v].push_back({u, weight});
    }

    void primMST() {
        vector<int> key(V, INT_MAX);
        vector<int> parent(V, -1);
        vector<bool> inMST(V, false);

        priority_queue<Node, vector<Node>, greater<Node>> pq;
        key[0] = 0;
        pq.push({0, 0});

        while (!pq.empty()) {
            int u = pq.top().vertex;
            pq.pop();
            inMST[u] = true;

            for (const auto &edge : adjList[u]) {
                int v = edge.dest;
                int weight = edge.weight;
                if (!inMST[v] && weight < key[v]) {
                    key[v] = weight;
                    pq.push({v, key[v]});
                    parent[v] = u;
                }
            }
        }

        printMST(parent);
    }

private:
    int V;
    vector<vector<Edge>> adjList;

    void printMST(const vector<int> &parent) {
        cout << "Edge \tWeight\n";
        for (int i = 1; i < V; ++i) {
            cout << parent[i] << " - " << i << "\t" << findWeight(parent[i], i) << endl;
        }
    }

    int findWeight(int u, int v) {
        for (const auto &edge : adjList[u]) {
            if (edge.dest == v)
                return edge.weight;
        }
        return 0;
    }
};

int main() {
    int V, E;
    cout << "Enter the number of vertices: ";
    cin >> V;

    cout << "Enter the number of edges: ";
    cin >> E;

    Graph graph(V);

    cout << "Enter the edges (u v weight):\n";
    for (int i = 0; i < E; ++i) {
        int u, v, weight;
        cin >> u >> v >> weight;
        graph.addEdge(u, v, weight);
    }

    cout << "\nMinimum Spanning Tree using Prim's Algorithm:\n";
    graph.primMST();

    return 0;
}
